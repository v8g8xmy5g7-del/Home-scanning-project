<body>

  <div class="container">

    <h1>Home Inventory</h1>

    <div id="scanner-container"></div>

    <div class="actions">

      <button id="startScan">Start Scanner</button>

      <button id="stopScan" disabled>Stop Scanner</button>

    </div>

    <hr/>

    <h3>Add Item</h3>

    <input type="text" id="barcode" placeholder="Barcode / QR Code" />

    <input type="text" id="name" placeholder="Item Name" />

    <input type="number" id="quantity" placeholder="Quantity" value="1" min="1" />

    <button id="addItem">Add to Inventory</button>

    <div id="search"><input type="text" id="searchInput" placeholder="Search items..." /></div>

    <div class="count">Total Items: <strong id="totalCount">0</strong></div>

    <div id="inventory"></div>

  </div>


  <script>

    let inventory = JSON.parse(localStorage.getItem('homeInventory')) || {};

    let scanner = null;


    document.getElementById('startScan').onclick = async () => {

      if (scanner) return;

      try {

        scanner = await Dynamsoft.DBR.BarcodeScanner.createInstance();

        scanner.setUIElement(document.getElementById('scanner-container'));

        scanner.onFrameRead = results => {

          if (results.length > 0) {

            const barcode = results[0].barcodeText;

            document.getElementById('barcode').value = barcode;

            stopScanner();

            alert(`Scanned: ${barcode}\nEnter name and add.`);

          }

        };

        scanner.onPlayed = () => {

          document.getElementById('startScan').disabled = true;

          document.getElementById('stopScan').disabled = false;

        };

        await scanner.open();

      } catch (e) {

        alert('Camera blocked — use phone + allow camera');

      }

    };


    document.getElementById('stopScan').onclick = () => {

      if (scanner) { scanner.close(); scanner = null; }

      document.getElementById('startScan').disabled = false;

      document.getElementById('stopScan').disabled = true;

    };


    document.getElementById('addItem').onclick = () => {

      const barcode = document.getElementById('barcode').value.trim();

      const name = document.getElementById('name').value.trim();

      const qty = parseInt(document.getElementById('quantity').value) || 1;

      if (!barcode || !name) { alert("Fill barcode and name"); return; }

      if (!inventory[barcode]) inventory[barcode] = {name, quantity: 0};

      inventory[barcode].quantity += qty;

      saveAndRender();

      clearInputs();

    };


    window.adjustQty = (bc, d) => {

      inventory[bc].quantity += d;

      if (inventory[bc].quantity <= 0) delete inventory[bc];

      saveAndRender();

    };

    window.removeItem = bc => {

      if (confirm(`Delete ${inventory[bc].name}?`)) { delete inventory[bc]; saveAndRender(); }

    };


    function saveAndRender() {

      localStorage.setItem('homeInventory', JSON.stringify(inventory));

      render();

    }

    function render() {

      const list = document.getElementById('inventory');

      const search = document.getElementById('searchInput').value.toLowerCase();

      const items = Object.entries(inventory)

        .filter(([bc, it]) => it.name.toLowerCase().includes(search) || bc.includes(search))

        .sort((a,b) => a[1].name.localeCompare(b[1].name));

      list.innerHTML = items.map(([bc, it]) => `

        <div class="item">

          <div><strong>${it.name}</strong><br><small>${bc}</small></div>

          <div style="text-align:right">

            <strong>${it.quantity}</strong><br>

            <button onclick="adjustQty('${bc}',1)">+1</button>

            <button onclick="adjustQty('${bc}',-1)">-1</button>

            <button style="background:#e74c3c" onclick="removeItem('${bc}')">Remove</button>

          </div>

        </div>`).join('');

      document.getElementById('totalCount').textContent = items.reduce((s,[_,it])=>s+it.quantity,0);

    }

    function clearInputs() {

      document.getElementById('barcode').value = '';

      document.getElementById('name').value = '';

      document.getElementById('quantity').value = '1';

    }


    document.getElementById('searchInput').addEventListener('input', render);

    document.getElementById('name').addEventListener('keypress', e => { if (e.key === 'Enter') document.getElementById('addItem').click(); });

    render();

  </script>
